class Solution {
    int getMinDiff(int[] arr, int n, int k) {
        // code here
        Arrays.sort(arr);
        // Maximum possible height difference
        int min=arr[0];
        int max = arr[n-1];
        int ans=max-min;
        for(int i=1;i<n;i++){
            if(arr[i]-k<0){continue;}
            min = Math.min(arr[0]+k,arr[i]-k);
            max = Math.max(arr[i-1]+k,arr[n-1]-k);
            ans = Math.min(ans,max-min);
        }
        return ans;
    }
}
